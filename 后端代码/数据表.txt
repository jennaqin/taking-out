CREATE TABLE table_user
(
userid INT PRIMARY KEY AUTO_INCREMENT,
userphone CHAR(11) NOT NULL UNIQUE,
password CHAR(35) NOT NULL,
address CHAR(50) NOT NULL,
name CHAR(10) NOT NULL,
icon CHAR(40) NOT NULL,
state SMALLINT NOT NULL
)ENGINE=INNODB DEFAULT CHARSET=UTF8  AUTO_INCREMENT=1;

CREATE TABLE table_mealtype
(
typeid INT PRIMARY KEY AUTO_INCREMENT,
typename VARCHAR(15) NOT NULL
)ENGINE=INNODB DEFAULT CHARSET=UTF8  AUTO_INCREMENT=1;

CREATE TABLE table_meal
(
mealid INT PRIMARY KEY AUTO_INCREMENT,
typeid INT NOT NULL,
mealname VARCHAR(10) NOT NULL,
mealprice FLOAT NOT NULL,
mealphoto VARCHAR(150) NOT NULL,
mealstate TINYINT NOT NULL
)ENGINE=INNODB DEFAULT CHARSET=UTF8  AUTO_INCREMENT=1;

CREATE TABLE table_cart
(
cartid INT PRIMARY KEY AUTO_INCREMENT,
userid INT NOT NULL,
mealid INT NOT NULL,
orderid INT NOT NULL,
count TINYINT NOT NULL
)ENGINE=INNODB DEFAULT CHARSET=UTF8  AUTO_INCREMENT=1;

CREATE TABLE table_order
(
orderid INT PRIMARY KEY AUTO_INCREMENT,
userid INT NOT NULL,
tradingtime TIMESTAMP NOT NULL,
orderstate TINYINT NOT NULL,
totalprice FLOAT NOT NULL
)ENGINE=INNODB DEFAULT CHARSET=UTF8  AUTO_INCREMENT=1;
